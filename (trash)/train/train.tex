\begin{prb}
\begin{parts}
\item
Let $f\in C^3(\R)$.
If $0<f'(x)$, $0<f''(x)$, $0<f'''(x)\le f(x)$, then $f'(x)\le2f(x)$.
\item
Let $f\in C^2(\R)$.
If $f''(x)\ge f(x)$, then $f(x)\ge f(0)\cosh x+f'(0)\sinh x$.
\end{parts}
\end{prb}
